{ 
	\leavevmode% 
	\hbox{% 
		\begin{beamercolorbox}[wd=\paperwidth,ht=2.25ex,dp=1ex,center]{footline}% 
			\usebeamerfont{author in head/foot}
			\insertshortauthor 
			\hspace{8em}
			\insertshorttitle
			\hspace{8em}
			\hfill
			% \insertshortdate{}
			\insertframenumber
			% \inserttotalframenumber\hspace*{2ex} 
	\end{beamercolorbox}}% 
	\vskip0pt% 
}
